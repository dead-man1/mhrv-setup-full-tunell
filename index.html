🚀 چند Deployment برای سرعت بیشتر

هر جیمیل = یه Deployment جداگانه = ظرفیت بیشتر

1
چند مرورگر مختلف نصب کن
مثلاً Chrome، Brave، Firefox — هر کدوم برای یه جیمیل جداگانه
2
در هر مرورگر با یه جیمیل وارد script.google.com بشو
CodeFull.gs رو deploy کن و ID رو بگیر
3
اولین deploy — هشدار unsafe
وقتی اول می‌خوای دسترسی بدی، گوگل هشدار «This app isn't verified» میده — روی Advanced کلیک کن، بعد Go to ... (unsafe) و دسترسی بده.
4
همه ID ها رو به اپ اضافه کن
در بخش Deployment IDs هر ID رو در یه خط جداگانه بنویس — اپ به‌صورت round-robin استفاده می‌کنه.
این کد CodeFull.gs هست — نه Code.gs قبلی. حتماً از این نسخه استفاده کن که tunnel node رو پشتیبانی می‌کنه.
⚙️ تنظیمات — فقط این ۳ خط رو عوض کن
AUTH_KEY
رمز دلخواه خودت — هم اینجا هم توی اپ باید یکی باشه
TUNNEL_SERVER_URL
آدرس IP و پورت VPS هلند — مثلاً: http://1.2.3.4:8080
TUNNEL_AUTH_KEY
معمولاً همون رمز AUTH_KEY هست
⚠️ باگ کپی‌پیست: وقتی کد رو کپی می‌کنی، یه خط خالی اضافه ممکنه آخر فایل اضافه بشه. قبل از Deploy حتماً آخر فایل رو چک کن و خط اضافه رو پاک کن.

const AUTH_KEY        = "YOUR_AUTH_KEY";          // ← رمز دلخواه
const TUNNEL_SERVER_URL = "http://YOUR_VPS_IP:8080"; // ← آدرس VPS
const TUNNEL_AUTH_KEY = "YOUR_AUTH_KEY";          // ← همون رمز بالا
// ============================================================

const SKIP_HEADERS = {
  host:1, connection:1, "content-length":1,
  "transfer-encoding":1, "proxy-connection":1,
  "proxy-authorization":1, "priority":1, te:1,
};

function doPost(e) {
  try {
    var req = JSON.parse(e.postData.contents);
    if (req.k !== AUTH_KEY) return _json({ e:"unauthorized" });
    if (req.t) return _doTunnel(req);
    if (Array.isArray(req.q)) return _doBatch(req.q);
    return _doSingle(req);
  } catch (err) { return _json({ e:String(err) }); }
}

function _doTunnel(req) {
  if (req.t === "batch") return _doTunnelBatch(req);
  var payload = { k: TUNNEL_AUTH_KEY };
  switch (req.t) {
    case "connect":
      payload.op="connect"; payload.host=req.h; payload.port=req.p; break;
    case "data":
      payload.op="data"; payload.sid=req.sid;
      if (req.d) payload.data=req.d; break;
    case "close":
      payload.op="close"; payload.sid=req.sid; break;
    default: return _json({ e:"unknown tunnel op: "+req.t });
  }
  var resp = UrlFetchApp.fetch(TUNNEL_SERVER_URL+"/tunnel", {
    method:"post", contentType:"application/json",
    payload:JSON.stringify(payload),
    muteHttpExceptions:true, followRedirects:true,
  });
  if (resp.getResponseCode() !== 200)
    return _json({ e:"tunnel node HTTP "+resp.getResponseCode() });
  return ContentService.createTextOutput(resp.getContentText())
    .setMimeType(ContentService.MimeType.JSON);
}

function _doTunnelBatch(req) {
  var payload = { k:TUNNEL_AUTH_KEY, ops:req.ops||[] };
  var resp = UrlFetchApp.fetch(TUNNEL_SERVER_URL+"/tunnel/batch", {
    method:"post", contentType:"application/json",
    payload:JSON.stringify(payload),
    muteHttpExceptions:true, followRedirects:true,
  });
  if (resp.getResponseCode() !== 200)
    return _json({ e:"tunnel batch HTTP "+resp.getResponseCode() });
  return ContentService.createTextOutput(resp.getContentText())
    .setMimeType(ContentService.MimeType.JSON);
}

function _doSingle(req) {
  if (!req.u||typeof req.u!=="string"||!req.u.match(/^https?:\/\//i))

    b:Utilities.base64Encode(resp.getContent()) });
}

apt update -y && apt upgrade -y

systemctl enable docker && systemctl start docker

ufw --force reset
ufw default deny incoming
ufw default allow outgoing
ufw allow ${SSH_PORT}/tcp
ufw allow ${PORT}/tcp

docker rm -f tunnel-node 2>/dev/null || true
docker pull ghcr.io/therealaleph/mhrv-tunnel-node:latest

  --name tunnel-node \
  --restart always \
  -p ${PORT}:${PORT} \

  -e PORT=${PORT} \
  ghcr.io/therealaleph/mhrv-tunnel-node:latest

⚡ بهینه‌سازی سرعت
  • حداقل ۳-۴ Deployment ID از جیمیل‌های مختلف اضافه کن
  • از بخش SNI pool، ۲-۳ آدرس با کمترین پینگ انتخاب کن
  • دکمه Auto-detect برای Google IP بزن تا بهترین IP پیدا بشه
  • Connection mode رو روی VPN (TUN) بذار که همه اپ‌ها cover بشن
⚠️ محدودیت‌های شناخته‌شده
WebSocket کار نمی‌کنه — ChatGPT streaming، Discord voice
ویدیو سنگین سهمیه دارد — برای یوتوب ۱۰۸۰p از VPN دیگه استفاده کن
User-Agent ثابته — بعضی سایت‌ها CAPTCHA نشون می‌دن
🔧 عیب‌یابی
  • Relay timeout: Apps Script درست deploy نشده — دوباره deploy کن
  • Certificate error: دکمه Install MITM Certificate رو بزن
  • تلگرام وصل نمیشه: از SOCKS5 proxy در تلگرام استفاده کن — 127.0.0.1:8086
  • VPS وصل نمیشه: ufw allow SSH_PORT/tcp رو اجرا کن

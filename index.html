🚀 چند Deployment برای سرعت بیشتر
1
چند مرورگر نصب کن
Chrome، Brave، Firefox — هر کدوم برای یه جیمیل جداگانه.
2
در هر مرورگر با یه جیمیل وارد script.google.com بشو
CodeFull.gs رو deploy کن و ID رو بگیر.
3
هشدار «unsafe» — نگران نباش
روی Advanced کلیک کن ← Go to ... (unsafe) ← دسترسی بده. این برای اولین deploy هر جیمیل نرمالیه.
4
همه ID ها رو به اپ اضافه کن
هر ID در یه خط — اپ به‌صورت round-robin استفاده می‌کنه و سرعت چند برابر میشه.
این CodeFull.gs هست — نه Code.gs قدیمی. حتماً از این نسخه استفاده کن که tunnel node رو پشتیبانی می‌کنه.
⚠️ باگ کپی‌پیست: بعد از Paste کردن، آخر فایل رو چک کن — اگه خط خالی اضافه بود پاکش کن وگرنه خطای Syntax میگیری.
⚙️ فقط این ۳ خط رو تغییر بده
AUTH_KEY
رمز دلخواه — هم اینجا هم توی اپ باید یکی باشه
TUNNEL_SERVER_URL
IP و پورت VPS — مثلاً: http://1.2.3.4:8080
TUNNEL_AUTH_KEY
معمولاً همون رمز AUTH_KEY هست
CodeFull.gs — کپی کن در Apps Script Paste کن

✅ فقط یه بار اجرا کن — همه چیز خودکار نصب میشه. نیازی به ساخت فایل جداگانه نیست.
⚙️ قبل از نصب — این ۲ مقدار رو تنظیم کن
TUNNEL_AUTH_KEY — رمز tunnel
SSH_PORT — پورت SSH سرورت
💡 بعد از تنظیم، کد پایین خودکار آپدیت میشه — فقط کپی کن و در ترمینال بزن.
🔧 نصب کامل tunnel-node — یک بار اجرا کن 
#!/bin/bash
# ══════════════════════════════════════════════
#  mhrv Tunnel Node — نصب کامل
#  Ubuntu 22.04 / Debian
# ══════════════════════════════════════════════

AUTH_KEY="YOUR_AUTH_KEY"   # ← رمز tunnel
SSH_PORT=22                 # ← پورت SSH
PORT=8080                   # پورت tunnel (تغییر نده)

echo ""
echo "┌─────────────────────────────────────┐"
echo "│  ۱ / ۵  آپدیت سیستم...             │"
echo "└─────────────────────────────────────┘"
export DEBIAN_FRONTEND=noninteractive
apt update -y && apt upgrade -y && \
echo "✅ آپدیت موفق" || { echo "❌ خطا در آپدیت"; exit 1; }

echo ""
echo "┌─────────────────────────────────────┐"
echo "│  ۲ / ۵  نصب Docker و ابزارها...    │"
echo "└─────────────────────────────────────┘"
apt install -y docker.io curl ufw && \
systemctl enable docker && systemctl start docker && \
echo "✅ Docker نصب شد" || { echo "❌ خطا در نصب Docker"; exit 1; }

echo ""
echo "┌─────────────────────────────────────┐"
echo "│  ۳ / ۵  تنظیم فایرول...            │"
echo "└─────────────────────────────────────┘"
ufw --force reset
ufw default deny incoming
ufw default allow outgoing
ufw allow ${SSH_PORT}/tcp && echo "✅ پورت SSH ${SSH_PORT} باز شد"
ufw allow ${PORT}/tcp     && echo "✅ پورت Tunnel ${PORT} باز شد"
ufw --force enable        && echo "✅ فایرول فعال شد"

echo ""
echo "┌─────────────────────────────────────┐"
echo "│  ۴ / ۵  دانلود image رسمی...        │"
echo "└─────────────────────────────────────┘"
docker rm -f tunnel-node 2>/dev/null || true
docker pull ghcr.io/therealaleph/mhrv-tunnel-node:latest && \
echo "✅ Image دانلود شد" || { echo "❌ خطا در دانلود image"; exit 1; }

echo ""
echo "┌─────────────────────────────────────┐"
echo "│  ۵ / ۵  راه‌اندازی سرویس...        │"
echo "└─────────────────────────────────────┘"
docker run -d \
  --name tunnel-node \
  --restart always \
  -p ${PORT}:${PORT} \
  -e TUNNEL_AUTH_KEY="${AUTH_KEY}" \
  -e PORT=${PORT} \
  ghcr.io/therealaleph/mhrv-tunnel-node:latest && \
echo "✅ Container اجرا شد"

sleep 5
echo ""
echo "┌─────────────────────────────────────┐"
echo "│  تست نهایی...                       │"
echo "└─────────────────────────────────────┘"
curl -sf http://localhost:${PORT}/health && \
echo "✅ سرویس OK — آماده استفاده‌ست!" || \
echo "❌ سرویس جواب نمیده — لاگ: docker logs tunnel-node"

echo ""
VPS_IP=$(curl -4s ifconfig.me 2>/dev/null)
echo "══════════════════════════════════════════"
echo "  ✅ نصب کامل شد"
echo "  📌 URL برای CodeFull.gs:"
echo "     http://${VPS_IP}:${PORT}"
echo "  🔑 AUTH_KEY: ${AUTH_KEY}"
echo "══════════════════════════════════════════"
🔍 چک کامل وضعیت VPS 
echo ""
echo "┌─────────────────────────────────────┐"
echo "│  وضعیت Docker Container             │"
echo "└─────────────────────────────────────┘"
docker ps --format "  نام: {{.Names}}  |  وضعیت: {{.Status}}  |  پورت: {{.Ports}}"

echo ""
echo "┌─────────────────────────────────────┐"
echo "│  تست Health                         │"
echo "└─────────────────────────────────────┘"
curl -sf http://localhost:8080/health && echo " ✅ سرویس OK" || echo " ❌ سرویس جواب نمیده"

echo ""
echo "┌─────────────────────────────────────┐"
echo "│  تست اتصال Tunnel                   │"
echo "└─────────────────────────────────────┘"
RESULT=$(curl -s -X POST http://localhost:8080/tunnel \
  -H "Content-Type: application/json" \
  -d '{"k":"YOUR_AUTH_KEY","op":"connect","host":"www.google.com","port":443}')
echo "$RESULT" | grep -q '"sid"' && echo " ✅ Tunnel OK — اتصال برقرار شد" || echo " ❌ Tunnel خطا: $RESULT"

echo ""
echo "┌─────────────────────────────────────┐"
echo "│  IP عمومی VPS                       │"
echo "└─────────────────────────────────────┘"
echo "  $(curl -4s ifconfig.me)"

echo ""
echo "┌─────────────────────────────────────┐"
echo "│  وضعیت فایرول                       │"
echo "└─────────────────────────────────────┘"
ufw status numbered

echo ""
echo "┌─────────────────────────────────────┐"
echo "│  مصرف منابع                         │"
echo "└─────────────────────────────────────┘"
docker stats tunnel-node --no-stream --format "  CPU: {{.CPUPerc}}  |  RAM: {{.MemUsage}}"

echo ""
echo "┌─────────────────────────────────────┐"
echo "│  آخرین لاگ‌ها                       │"
echo "└─────────────────────────────────────┘"
docker logs tunnel-node --tail 10
🔄 آپدیت به آخرین نسخه 
echo "🔄 آپدیت tunnel-node..."
docker pull ghcr.io/therealaleph/mhrv-tunnel-node:latest && \
echo "✅ Image جدید دانلود شد" || { echo "❌ خطا در دانلود"; exit 1; }

docker rm -f tunnel-node 2>/dev/null
docker run -d \
  --name tunnel-node \
  --restart always \
  -p 8080:8080 \
  -e TUNNEL_AUTH_KEY="YOUR_AUTH_KEY" \
  -e PORT=8080 \
  ghcr.io/therealaleph/mhrv-tunnel-node:latest

sleep 4
curl -sf http://localhost:8080/health && \
echo "✅ آپدیت موفق — سرویس OK" || \
echo "❌ خطا — بزن: docker logs tunnel-node"

⚡ بهینه‌سازی سرعت
  • حداقل ۳-۴ Deployment ID از جیمیل‌های مختلف اضافه کن
  • از بخش SNI pool، آدرس‌هایی با کمترین پینگ انتخاب کن
  • دکمه Auto-detect برای Google IP بزن
  • Connection mode رو روی VPN (TUN) بذار
⚠️ محدودیت‌های شناخته‌شده
WebSocket کار نمی‌کنه — ChatGPT streaming، Discord voice
ویدیو سنگین سهمیه دارد — برای ۱۰۸۰p از چند Deployment استفاده کن
User-Agent ثابته — بعضی سایت‌ها CAPTCHA نشون می‌دن
🔧 عیب‌یابی
  • Relay timeout: Apps Script درست deploy نشده
  • Certificate error: دکمه Install MITM Certificate رو بزن
  • تلگرام: Settings → Proxy → SOCKS5 → 127.0.0.1:8086
  • VPS قفل SSH: فایرول پورت SSH رو بسته — از پنل هاستینگ Reboot کن
